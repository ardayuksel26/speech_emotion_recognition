% ==========================================
% 1. FONTS AND ENCODING
% ==========================================
\usepackage[utf8]{inputenc}

% ==========================================
% 3. LINE SPACING AND PARAGRAPHS
% ==========================================
\usepackage{setspace}
\onehalfspacing % Enforces 1.5 line spacing for main text

\usepackage{indentfirst} % Indents the first paragraph of every section
\setlength{\parindent}{1cm} % 1 cm paragraph indentation
\setlength{\parskip}{0pt} % No extra vertical space between paragraphs

% Lists (bullets/numbered) without extra vertical spacing
\usepackage{enumitem}
\setlist{nosep} 

% ==========================================
% 4. PAGINATION AND HEADERS
% ==========================================
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{} % Clear default headers/footers
\renewcommand{\headrulewidth}{0pt} % Remove horizontal line at the top
% Page numbers in Arabic numerals: 2cm from top, 2cm from right
\fancyhead[R]{\thepage}
\patchcmd{\fancyhead}{\rlap}{\hbox}{}{} % Ensure header alignment (sometimes needed)
\setlength{\headsep}{0.5cm} % 3cm top margin - 0.5cm (headsep) - 0.5cm (pagenum height?)
% Actually the simplest way is geometry + fancyhdr as is, usually top margin is from page edge.

% ==========================================
% 5. SECTION HEADINGS FORMATTING
% ==========================================
\usepackage{titlesec}

% Force every Level 1 section to start on a new page
\newcommand{\sectionbreak}{\clearpage}

% Remove periods after section numbers (e.g., "1", not "1.")
\renewcommand{\thesection}{\arabic{section}}
\renewcommand{\thesubsection}{\thesection.\arabic{subsection}}
\renewcommand{\thesubsubsection}{\thesubsection.\arabic{subsubsection}}

% Level 1 (\section): 14pt, Bold, Centered, All Caps
\titleformat{\section}[block]
  {\normalfont\fontsize{14}{17}\bfseries\centering}
  {\thesection \ \ } % Two spaces between number and title
  {0pt}
  {\MakeUppercase}
\titlespacing*{\section}{0pt}{1.5\baselineskip}{1.5\baselineskip}

% Level 2 (\subsection): 12pt, Bold, Left-aligned, Title Case
\titleformat{\subsection}[block]
  {\normalfont\fontsize{12}{15}\bfseries}
  {\thesubsection \ \ } % Two spaces between number and title
  {0pt}
  {\TitleCase}
\titlespacing*{\subsection}{0pt}{1.5\baselineskip}{1.5\baselineskip}

% Level 3 (\subsubsection): 12pt, Bold, Left-aligned, Sentence Case
\titleformat{\subsubsection}[block]
  {\normalfont\fontsize{12}{15}\bfseries}
  {\thesubsubsection \ \ } % Two spaces between number and title
  {0pt}
  {}
\titlespacing*{\subsubsection}{0pt}{1.5\baselineskip}{0pt} % No extra space AFTER heading
